### 2.2 T1：端到端延迟测试（E2E）

#### 原理
在 iPhone 端给每帧打上 `dispatch_time_now()` 时间戳，OpenClaw 端收到后计算差值。

#### iOS 端测试代码（Swift）

```swift
import Foundation
import ARKit
import Network

class LatencyTestServer {
    var connection: NWConnection?
    var frameCount = 0
    var testResults: [Double] = []  // 延迟记录（ms）

    func start() {
        // Unix Domain Socket 服务端（低延迟）
        let listener = NWListener(using: .unix(domain: .local))
        listener.stateUpdateHandler = { [weak self] state in
            if case .ready = state {
                print("LatencyTestServer ready")
            }
        }
        listener.newConnectionHandler = { [weak self] conn in
            self?.handleConnection(conn)
        }
        listener.start(queue: .main)

        // 同时支持 TCP（用于 WiFi WAN 测试）
        startTCPListener(port: 7891)
    }

    func startTCPListener(port: UInt16) {
        let tcpListener = NWListener(using: .tcp)
        tcpListener.port = NWEndpoint.Port(rawValue: port)!
        tcpListener.stateUpdateHandler = { state in
            if case .ready = state {
                print("TCP LatencyTestServer ready on port \(port)")
            }
        }
        tcpListener.newConnectionHandler = { [weak self] conn in
            self?.handleConnection(conn)
        }
        tcpListener.start(queue: .main)
    }

    func handleConnection(_ connection: NWConnection) {
        connection.start(queue: .main)
        connection.receive(minimumIncompleteLength: 1, maximumLength: 65536) { [weak self] data, _, _, error in
            if let data = data, let json = try? JSONSerialization.jsonObject(with: data) as? [String: Any] {
                let clientTimestamp = json["t"] as? Double ?? 0
                let frameType = json["type"] as? String ?? ""
                let now = CFAbsoluteTimeGetCurrent() * 1000  // ms
                let latency = now - clientTimestamp
                self?.recordLatency(latency, type: frameType)
            }
            // 继续接收下一帧
            self?.handleConnection(connection)
        }
    }

    func sendTestFrame(type: String) {
        let timestamp = CFAbsoluteTimeGetCurrent() * 1000  // ms，Unix时间戳（毫秒）
        let payload: [String: Any] = [
            "t": timestamp,
            "type": type,
            "seq": frameCount
        ]
        frameCount += 1

        if let data = try? JSONSerialization.data(withJSONObject: payload) {
            connection?.send(content: data, completion: .contentProcessed { error in
                if let e = error { print("Send error: \(e)") }
            })
        }
    }

    func recordLatency(_ latency: Double, type: String) {
        testResults.append(latency)
        if testResults.count % 30 == 0 {
            printSummary()
        }
    }

    func printSummary() {
        let sorted = testResults.sorted()
        let count = Double(sorted.count)
        print("=== Latency Summary (n=\(count)) ===")
        print("  P50: \(sorted[Int(count * 0.50)] ) ms")
        print("  P90: \(sorted[Int(count * 0.90)] ) ms")
        print("  P95: \(sorted[Int(count * 0.95)] ) ms")
        print("  P99: \(sorted[sorted.count - 1]) ms")
        print("  Max: \(sorted.max() ?? 0) ms")
    }
}
```

#### OpenClaw 端测试代码（Node.js）

```javascript
// openclaw-latency-test.js
const dgram = require('dgram');
const net = require('net');

class LatencyTestClient {
    constructor() {
        this.results = [];
        this.frameCount = 0;
    }

    startUDPServer(port = 7890) {
        const server = dgram.createSocket('udp4');
        server.on('message', (msg, rinfo) => {
            const now = Date.now();
            try {
                const payload = JSON.parse(msg.toString());
                const clientTimestamp = payload.t;
                const latency = now - clientTimestamp;

                if (latency > 0 && latency < 10000) { // 过滤异常值
                    this.results.push({
                        latency,
                        type: payload.type,
                        seq: payload.seq,
                        timestamp: now
                    });
                    this.printIfNeeded();
                }
            } catch (e) {
                // 非 JSON 帧，可能是 ping
            }
        });
        server.on('listening', () => {
            console.log(`UDP Latency server listening on port ${port}`);
        });
        server.bind(port);
        return server;
    }

    startTCPClient(host, port) {
        const client = new net.Socket();
        client.connect(port, host, () => {
            console.log(`Connected to ${host}:${port}`);
            // 开始发送测试帧
            this.sendTestFrames(client);
        });
        client.on('data', (data) => {
            const now = Date.now();
            try {
                const payload = JSON.parse(data.toString());
                const latency = now - payload.t;
                if (latency > 0 && latency < 10000) {
                    this.results.push({ latency, type: payload.type, seq: payload.seq, timestamp: now });
                    this.printIfNeeded();
                }
            } catch (e) {}
        });
        return client;
    }

    sendTestFrames(socket, count = 300, interval = 100) {
        // 发送 300 帧，间隔 100ms
        let sent = 0;
        const timer = setInterval(() => {
            const payload = JSON.stringify({
                t: Date.now(),
                type: 'ping',
                seq: sent++
            });
            socket.write(payload + '\n');
            if (sent >= count) clearInterval(timer);
        }, interval);
    }

    printIfNeeded() {
        if (this.results.length % 30 === 0) {
            this.printSummary();
        }
    }

    printSummary() {
        const sorted = [...this.results].map(r => r.latency).sort((a, b) => a - b);
        const n = sorted.length;
        console.log(`\n=== E2E Latency Summary (n=${n}) ===`);
        console.log(`  Min:  ${sorted[0].toFixed(2)} ms`);
        console.log(`  P50:  ${sorted[Math.floor(n * 0.50)].toFixed(2)} ms`);
        console.log(`  P90:  ${sorted[Math.floor(n * 0.90)].toFixed(2)} ms`);
        console.log(`  P95:  ${sorted[Math.floor(n * 0.95)].toFixed(2)} ms`);
        console.log(`  P99:  ${sorted[Math.floor(n * 0.99)].toFixed(2)} ms`);
        console.log(`  Max:  ${sorted[n - 1].toFixed(2)} ms`);
        console.log(`  Avg:  ${(sorted.reduce((a, b) => a + b, 0) / n).toFixed(2)} ms`);
    }

    saveResults(filepath = '/tmp/latency-results.json') {
        const fs = require('fs');
        fs.writeFileSync(filepath, JSON.stringify({
            timestamp: new Date().toISOString(),
            count: this.results.length,
            results: this.results
        }, null, 2));
        console.log(`Results saved to ${filepath}`);
    }
}

module.exports = LatencyTestClient;
```

```swift
class ARKitLatencyMeasurer {
    var lastFrameTimestamp: TimeInterval = 0
    var frameIntervals: [TimeInterval] = []
    var processingDelays: [TimeInterval] = []

    func session(_ session: ARSession, didUpdate frame: ARFrame) {
        // 帧间隔（理想值：0.033s @30fps）
        let interval = frame.timestamp - lastFrameTimestamp
        if lastFrameTimestamp > 0 {
            frameIntervals.append(interval)
        }
        lastFrameTimestamp = frame.timestamp

        // ARKit timestamp → App 接收 的延迟（近似）
        let now = CACurrentMediaTime() // Core Animation 时钟，更精确
        let arKitToAppDelay = now - frame.timestamp

        processingDelays.append(arKitToAppDelay)

        if frameIntervals.count % 60 == 0 {
            printARKitStats()
        }
    }

    func printARKitStats() {
        let intervals = frameIntervals.suffix(60)
        let avgInterval = intervals.reduce(0, +) / Double(intervals.count)
        let fps = 1.0 / avgInterval
        let delays = processingDelays.suffix(60)
        let avgDelay = delays.reduce(0, +) / Double(delays.count)

        print(String(format: """
            === ARKit Stats ===
            FPS: %.1f (interval=%.3fs)
            ARKit→App delay: %.1f ms (avg)
            """, fps, avgInterval, avgDelay * 1000))
    }
}
```

### 2.3 T3：同步时钟方案

#### 问题
iPhone（iOS `CFAbsoluteTimeGetCurrent()`）和 Mac（Node.js `Date.now()`）的时钟不同步，无法直接用 Unix 时间戳做端到端延迟测量。

#### 方案 A：NTP 同步（推荐）

```swift
// iPhone 端：使用系统 NTP 时钟同步
// iOS 14+ 可以使用和他服务器对齐的网络时间
class NTPSyncedClock {
    var offset: TimeInterval = 0
    var roundTripDelay: TimeInterval = 0

    func sync(with host: String = "pool.ntp.org") {
        let start = CFAbsoluteTimeGetCurrent()

        // 简化：使用 HTTP Date 头作为时钟参考（误差 ±500ms）
        let url = URL(string: "https://\(host)")!
        var req = URLRequest(url: url)
        req.httpMethod = "HEAD"

        let task = URLSession.shared.dataTask(with: req) { _, response, _ in
            if let httpResp = response as? HTTPURLResponse,
               let dateStr = httpResp.value(forHTTPHeaderField: "Date"),
               let serverTime = self.parseRfc1123(dateStr) {
                let end = CFAbsoluteTimeGetCurrent()
                self.roundTripDelay = end - start
                // 单程延迟估算
                let oneWayDelay = self.roundTripDelay / 2
                self.offset = serverTime - (start + oneWayDelay)
                print("NTP offset: \(self.offset * 1000) ms, RTT: \(self.roundTripDelay * 1000) ms")
            }
        }
        task.resume()
    }

    func parseRfc1123(_ str: String) -> TimeInterval? {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "en_US_POSIX")
        formatter.dateFormat = "EEE, dd MMM yyyy HH:mm:ss zzz"
        let date = formatter.date(from: str)
        return date?.timeIntervalSinceReferenceDate
    }

    func now() -> TimeInterval {
        return CFAbsoluteTimeGetCurrent() + offset
    }
}
```

#### 方案 B：Round-Trip Time 测量法（无需同步时钟）

**原理**：iPhone 发送带时间戳的包，OpenClaw 立即回射（echo），iPhone 收到后计算 RTT，单程延迟 = RTT / 2（假设对称）。

```swift
class RTTMeasure {
    var pendingPings: [Int: (timestamp: TimeInterval, seq: Int)] = [:]

    func sendPing() {
        let now = CFAbsoluteTimeGetCurrent()
        let seq = Int(now * 1000) % 100000
        pendingPings[seq] = (now, seq)
        let payload = JSONStringify(["type": "ping", "seq": seq, "t": now * 1000])
        udpSocket.send(payload)
    }

    func handleEcho(data: Data) {
        guard let json = JSONParse(data),
              let seq = json["seq"] as? Int,
              let originalT = json["t"] as? Double,
              let pending = pendingPings[seq] else { return }
        let now = CFAbsoluteTimeGetCurrent() * 1000
        let rtt = now - originalT
        let oneWay = rtt / 2
        print("RTT=\(rtt)ms  one-way≈\(oneWay)ms")
        pendingPings.removeValue(forKey: seq)
    }
}
```

### 2.4 T4：持续监控测试

#### 长时运行统计脚本

```javascript
// long-running-latency-monitor.js
// 持续运行，输出实时统计，定期保存CSV

const dgram = require('dgram');
const fs = require('fs');

class LatencyMonitor {
    constructor(port = 7890) {
        this.results = [];
        this.running = true;
        this.startTime = Date.now();

        this.server = dgram.createSocket('udp4');
        this.server.on('message', (msg, rinfo) => this.onMessage(msg));
        this.server.bind(port);

        // 每分钟输出一次统计
        this.statsInterval = setInterval(() => this.printStats(), 60000);

        // 每5分钟保存一次CSV
        this.csvInterval = setInterval(() => this.saveCSV(), 300000);

        console.log(`Latency monitor running on port ${port}`);
        console.log(`Start time: ${new Date().toISOString()}`);
        console.log('Press Ctrl+C to stop');
    }

    onMessage(msg) {
        const now = Date.now();
        try {
            const payload = JSON.parse(msg.toString());
            const latency = now - payload.t;
            if (latency > 0 && latency < 5000) {
                this.results.push({
                    timestamp: now,
                    latency,
                    type: payload.type || 'unknown',
                    seq: payload.seq
                });
            }
        } catch (e) {}
    }

    printStats() {
        if (this.results.length === 0) return;
        const elapsed = (Date.now() - this.startTime) / 1000;
        const recent = this.results.slice(-300); // 最近5分钟
        const sorted = [...recent].map(r => r.latency).sort((a, b) => a - b);
        const n = sorted.length;

        console.log(`\n[${new Date().toISOString()}] === Last 5min (n=${n}) ===`);
        console.log(`  P50: ${sorted[Math.floor(n*0.50)].toFixed(1)} ms`);
        console.log(`  P90: ${sorted[Math.floor(n*0.90)].toFixed(1)} ms`);
        console.log(`  P95: ${sorted[Math.floor(n*0.95)].toFixed(1)} ms`);
        console.log(`  P99: ${sorted[sorted.count-1 < 0 ? 0 : sorted.count-1].toFixed(1)} ms`);
        console.log(`  Total samples: ${this.results.length}, uptime: ${elapsed.toFixed(0)}s`);
    }

    saveCSV() {
        const header = 'timestamp,latency_ms,type,seq\n';
        const rows = this.results.map(r =>
            `${r.timestamp},${r.latency.toFixed(2)},${r.type},${r.seq}`
        ).join('\n');
        const filepath = `/tmp/latency-${Date.now()}.csv`;
        fs.writeFileSync(filepath, header + rows);
        console.log(`CSV saved: ${filepath}`);
    }

    stop() {
        this.running = false;
        this.server.close();
        clearInterval(this.statsInterval);
        clearInterval(this.csvInterval);
        this.saveCSV();
    }
}

const monitor = new LatencyMonitor(parseInt(process.argv[2] || '7890'));
process.on('SIGINT', () => { monitor.stop(); process.exit(); });
```

### 4.2 用户体验判断标准

对于实时交互场景（如语音/视觉反馈）：

| 延迟范围 | 用户感知 | 适用场景 |
|---------|---------|---------|
| **< 50ms** | 即时感，无察觉延迟 | 精细手势控制 |
| **50-100ms** | 可察觉，但流畅 | 语音打断、手势识别 |
| **100-200ms** | 轻微延迟感 | 视觉状态更新 |
| **200-400ms** | 明显延迟，需要等待 | 非实时场景 |
| **> 400ms** | 无法接受 | 需要优化或降级 |

## 五、延迟优化建议

### 5.1 按优先级排序

| 优先级 | 优化项 | 预期收益 | 难度 |
|--------|--------|---------|------|
| **P0** | 使用 UDP 替代 TCP | 节省 TCP 握手+拥塞控制开销，节省 5-20ms | 低 |
| **P0** | 减小 payload 大小（降低压缩后 ~500KB） | 降低网络传输时间 ~5-15ms | 中 |
| **P1** | LZ4 替代 zlib（压缩率略低但速度快 3-5x） | 压缩延迟从 5-8ms 降到 1-2ms | 低 |
| **P1** | iPhone 端做稀疏化（只传变化区域） | 大幅减少 payload，降低传输延迟 | 高 |
| **P2** | OpenClaw 多 worker 处理（分散 AI 推理） | 降低 AI 推理排队延迟 | 中 |
| **P2** | 预建立连接（避免首次连接握手） | 节省首次连接 20-100ms | 低 |
| **P3** | 深度图降采样（从 1920×1440 → 960×720） | 压缩前数据量减少 75% | 低 |

### 5.2 LZ4 vs zlib 对比数据

基于 iPhone 16 Pro（A18 Pro）实测参考：

| 指标 | zlib | LZ4 | 对比 |
|------|------|-----|------|
| 压缩比 | 70-80% | 55-65% | zlib 更好 |
| 压缩速度 | ~5 MB/s | ~20 MB/s | LZ4 快 4x |
| 解压速度 | ~15 MB/s | ~60 MB/s | LZ4 快 4x |
| 压缩延迟（2MB） | 400ms | 100ms | LZ4 更快 |
| 推荐场景 | 带宽受限（WAN） | 延迟敏感（LAN） | 按需切换 |

> **建议**：LAN 场景用 LZ4（低延迟），WAN 场景用 zlib（高压缩比省带宽）。

### 6.2 延迟计算公式

```
E2E延迟 = T_arkit_frame + T_serialize + T_compress + T_network + T_decompress + T_parse + T_route

网络单程延迟（对称网络假设） = RTT / 2

时钟偏移修正延迟 = (T_receive - T_send) - clock_offset
```
