```python
#!/usr/bin/env python3
"""自动审核 §2.7 手部与执行器设计"""
import re

with open("harness/robot/night-build/reference/ROBOT-SOP.md") as f:
    lines = f.readlines()

section = lines[2282:2376]  # 0-indexed
issues = []

# 检查1：空表格单元格
for i, line in enumerate(section):
    if "|" in line:
        cells = [c.strip() for c in line.split("|")]
        for j, cell in enumerate(cells):
            if cell in ("", "—", "TBD", "-") and j > 0:
                issues.append({"type": "data_missing", "line": 2283+i+1, "description": f"表格空单元格（第{j}列）", "severity": "medium"})

# 检查2：来源标注
for i, line in enumerate(section):
    if re.search(r'\d+\s*(元|¥|\$|kg|DOF|GB|TB)', line) and not re.search(r'来源|source|参考|ref', line, re.IGNORECASE):
        # 包含具体数据但附近没有来源标注
        issues.append({"type": "possible_no_source", "line": 2283+i+1, "description": f"含数据但无来源: {line.strip()[:60]}", "severity": "low"})

# 检查3：重复段落
text = "".join(section)
# 找出重复出现的短语（>20字符，出现>1次）
phrases = re.findall(r'.{20,}', text)
seen = {}
for p in phrases:
    if p in seen:
        issues.append({"type": "duplicate", "line": None, "description": f"重复内容: {p[:50]}...", "severity": "low"})
    seen[p] = True

import json
result = {"task_id": "T-01-H01", "script_issues": len(issues), "issues": issues}
with open("harness/robot/night-build/output/2026-03-26/T-01-H01/script-result.json", "w") as f:
    json.dump(result, f, ensure_ascii=False, indent=2)
print(f"Script found {len(issues)} potential issues")
```
