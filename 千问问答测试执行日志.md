with sync_playwright() as p:
    browser = p.chromium.connect_over_cdp("http://127.0.0.1:18800")
    
    # 找到千问页面
    for page in browser.contexts[0].pages:
        if "qwen" in page.url.lower():
            # 点击聚焦
            page.click("textarea")
            # 模拟打字
            page.keyboard.type("请用一句话解释什么是人工智能？", delay=100)

            break
