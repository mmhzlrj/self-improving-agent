## 1. 概述

### 1.1 背景

小龙虾 P2P 网络旨在实现 **AI to AI + Human-in-the-loop** 的自主交易模式，在提升交易效率的同时，确保合规性和安全性。

### 1.2 设计目标

- **合规优先**：满足中国法律对 AI 生成内容、支付、数据隐私的合规要求
- **人机协作**：支持 Full Auto / Approval / Supervision 三种模式
- **安全可控**：异常检测 + 熔断机制确保系统稳定性
- **可审计**：完整日志记录支持事后追溯

```typescript
// 触发条件配置
interface InterventionTrigger {
  // 金额触发
  amountThreshold?: {
    min: number;           // 最小金额
    max?: number;          // 最大金额（可选）
    mode: InterventionMode;
  };
  
  // 风险等级触发
  riskLevel?: {
    levels: RiskLevel[];   // 触发风险等级
    mode: InterventionMode;
  };
  
  // 异常行为触发
  anomalyDetected?: {
    types: AnomalyType[];  // 异常类型
    mode: InterventionMode;
  };
  
  // 人工主动触发
  manualRequested?: {
    by: 'user' | 'system';
  };
  
  // 定时触发（定期人工审核）
  periodicReview?: {
    intervalHours: number;
    mode: InterventionMode;
  };
}

// 风险等级枚举
enum RiskLevel {
  LOW = 'low',
  MEDIUM = 'medium', 
  HIGH = 'high',
  CRITICAL = 'critical'
}

// 干预模式
enum InterventionMode {
  FULL_AUTO = 'full_auto',     // 无需介入
  APPROVAL = 'approval',       // 需要审批
  SUPERVISION = 'supervision'  // 需要监督
}

// 异常类型
enum AnomalyType {
  UNUSUAL_AMOUNT = 'unusual_amount',           // 金额异常
  UNUSUAL_FREQUENCY = 'unusual_frequency',      // 频率异常
  UNUSUAL_TIME = 'unusual_time',                // 时间异常

  ACCOUNT_CHANGE = 'account_change',            // 账户变更
  DEVICE_CHANGE = 'device_change',              // 设备变更
  LOCATION_CHANGE = 'location_change'           // 位置变更
}
```

## 4. AI 身份披露机制

### 4.1 披露原则

根据《生成式人工智能服务管理暂行办法》和《互联网信息服务深度合成管理规定》：

1. **明确标识**：所有 AI 生成内容必须明确标识为 AI 生成
2. **可追溯**：标识信息应可追溯到具体 AI Agent
3. **不欺骗**：不得冒充人类进行交易

## 8. 代码示例

### 8.1 核心类：模式管理器

```typescript
// mode-manager.ts
import { EventEmitter } from 'events';
import { AuditLogger } from './audit-logger';
import { AnomalyDetector } from './anomaly-detector';
import { CircuitBreaker } from './circuit-breaker';
import { AIDisclosure } from './ai-disclosure';

export class ModeManager extends EventEmitter {
  private auditLogger: AuditLogger;
  private anomalyDetector: AnomalyDetector;
  private circuitBreaker: CircuitBreaker;
  private aiDisclosure: AIDisclosure;
  
  // 模式配置
  private modeConfig: Map<string, InterventionTrigger> = new Map();
  
  constructor(
    private decisionEngine: DecisionEngine,
    private transactionEngine: TransactionEngine
  ) {
    super();
    this.auditLogger = new AuditLogger();
    this.anomalyDetector = new AnomalyDetector();
    this.circuitBreaker = new CircuitBreaker();
    this.aiDisclosure = new AIDisclosure();
    
    this.initializeModeConfig();
  }
  
  private initializeModeConfig() {
    // 金额触发配置
    this.modeConfig.set('amount', {
      amountThreshold: {
        min: 1000,
        mode: InterventionMode.APPROVAL
      },
      amountThreshold: {
        min: 5000,
        mode: InterventionMode.SUPERVISION
      }
    });
    
    // 风险等级触发
    this.modeConfig.set('risk', {
      riskLevel: {
        levels: [RiskLevel.HIGH, RiskLevel.CRITICAL],
        mode: InterventionMode.SUPERVISION
      }
    });
  }
  
  /**
   * 确定交易模式
   */
  async determineMode(transaction: Transaction): Promise<InterventionMode> {
    const riskAssessment = await this.decisionEngine.assessRisk(transaction);
    
    // 检查熔断器状态
    if (this.circuitBreaker.isOpen()) {
      // 熔断时强制进入审批模式
      return InterventionMode.APPROVAL;
    }
    
    // 根据金额确定模式
    if (transaction.amount >= 5000) {
      return InterventionMode.SUPERVISION;
    }
    
    if (transaction.amount >= 1000) {
      return InterventionMode.APPROVAL;
    }
    
    // 根据风险等级确定模式
    if (riskAssessment.level === RiskLevel.CRITICAL) {
      return InterventionMode.SUPERVISION;
    }
    
    if (riskAssessment.level === RiskLevel.HIGH) {
      return InterventionMode.APPROVAL;
    }
    
    // 检测异常
    const anomalies = await this.anomalyDetector.detect(transaction);
    if (anomalies.length > 0) {
      const highestSeverity = this.getHighestSeverity(anomalies);
      if (highestSeverity === 'critical') {
        return InterventionMode.SUPERVISION;
      }
      return InterventionMode.APPROVAL;
    }
    
    return InterventionMode.FULL_AUTO;
  }
  
  /**
   * 执行交易
   */
  async execute(transaction: Transaction): Promise<TransactionResult> {
    // 确定模式
    const mode = await this.determineMode(transaction);
    
    // 应用 AI 披露
    const disclosure = this.aiDisclosure.generate(transaction.agentId);
    
    // 记录审计日志
    await this.auditLogger.log({
      logType: LogType.DECISION,
      eventType: 'mode_determined',
      transaction: { transactionId: transaction.id, amount: transaction.amount },
      action: { operation: 'determine_mode', result: 'success' },
      risk: { riskLevel: transaction.riskLevel },
      compliance: { disclosureApplied: true }
    });
    
    switch (mode) {
      case InterventionMode.FULL_AUTO:
        return this.executeFullAuto(transaction, disclosure);
      
      case InterventionMode.APPROVAL:
        return this.executeApproval(transaction, disclosure);
      
      case InterventionMode.SUPERVISION:
        return this.executeSupervision(transaction, disclosure);
    }
  }
  
  private async executeFullAuto(
    transaction: Transaction, 
    disclosure: AIDisclosureInfo
  ): Promise<TransactionResult> {
    // AI 决策
    const decision = await this.decisionEngine.decide(transaction);
    
    // 风险检查
    const riskCheck = await this.checkRisk(transaction);
    if (!riskCheck.passed) {
      // 降级到审批模式
      return this.executeApproval(transaction, disclosure);
    }
    
    // 执行交易
    const result = await this.transactionEngine.execute(transaction, decision);
    
    // 记录日志
    await this.auditLogger.log({
      logType: LogType.FUND,
      eventType: 'transaction_completed',
      transaction: { transactionId: transaction.id, amount: transaction.amount },
      action: { operation: 'execute', result: 'success' },
      compliance: { disclosureApplied: true, complianceChecked: true }
    });
    
    return { ...result, mode: InterventionMode.FULL_AUTO, disclosure };
  }
  
  private async executeApproval(
    transaction: Transaction,
    disclosure: AIDisclosureInfo
  ): Promise<TransactionResult> {
    // 进入待审批状态
    const pending = await this.transactionEngine.setPending(transaction);
    
    // 推送给审批人
    await this.notifyApprover(transaction, pending);
    
    // 记录日志
    await this.auditLogger.log({
      logType: LogType.DECISION,
      eventType: 'awaiting_approval',
      transaction: { transactionId: transaction.id, amount: transaction.amount },
      action: { operation: 'pending', result: 'pending' },
      compliance: { disclosureApplied: true }
    });
    
    // 等待审批结果（通过回调或轮询）
    return new Promise((resolve) => {
      this.once(`approval:${transaction.id}`, async (approval: Approval) => {
        if (approval.approved) {
          const result = await this.transactionEngine.execute(transaction, approval.decision);
          
          await this.auditLogger.log({
            logType: LogType.FUND,
            eventType: 'transaction_approved',
            transaction: { transactionId: transaction.id },
            intervention: {
              mode: InterventionMode.APPROVAL,
              triggeredBy: 'human',
              approverId: approval.approverId,
              comments: approval.comments
            },
            compliance: { disclosureApplied: true, complianceChecked: true }
          });
          
          resolve({ ...result, mode: InterventionMode.APPROVAL, disclosure });
        } else {
          await this.transactionEngine.cancel(transaction);
          
          await this.auditLogger.log({
            logType: LogType.DECISION,
            eventType: 'transaction_rejected',
            intervention: {
              mode: InterventionMode.APPROVAL,
              triggeredBy: 'human',
              approverId: approval.approverId,
              comments: approval.comments
            }
          });
          
          resolve({ status: 'rejected', reason: approval.comments });
        }
      });
    });
  }
  
  private async executeSupervision(
    transaction: Transaction,
    disclosure: AIDisclosureInfo
  ): Promise<TransactionResult> {
    // 拆分交易为多个步骤
    const steps = await this.transactionEngine.createSteps(transaction);
    
    for (const step of steps) {
      // 推送给监督人
      await this.notifySupervisor(transaction, step);
      
      // 记录步骤开始
      await this.auditLogger.log({
        logType: LogType.DECISION,
        eventType: 'step_started',
        transaction: { transactionId: transaction.id },
        action: { operation: `step:${step.id}`, result: 'pending' }
      });
      
      // 等待监督结果
      const supervision = await this.waitForSupervision(transaction, step);
      
      if (supervision.action === 'cancel') {
        await this.transactionEngine.cancel(transaction);
        
        await this.auditLogger.log({
          logType: LogType.DECISION,
          eventType: 'supervision_canceled',
          intervention: {
            mode: InterventionMode.SUPERVISION,
            triggeredBy: 'human',
            comments: supervision.comments
          }
        });
        
        return { status: 'canceled', reason: supervision.comments };
      }
      
      if (supervision.action === 'rollback') {
        await this.transactionEngine.rollback(transaction, step);
        
        await this.auditLogger.log({
          logType: LogType.DECISION,
          eventType: 'supervision_rollback',
          intervention: {
            mode: InterventionMode.SUPERVISION,
            triggeredBy: 'human',
            comments: supervision.comments
          }
        });
        
        return { status: 'rolled_back', reason: supervision.comments };
      }
      
      // 执行当前步骤
      await this.transactionEngine.executeStep(transaction, step);
      
      // 记录步骤完成
      await this.auditLogger.log({
        logType: LogType.FUND,
        eventType: 'step_completed',
        transaction: { transactionId: transaction.id }
      });
    }
    
    return { status: 'completed', mode: InterventionMode.SUPERVISION, disclosure };
  }
  
  private async checkRisk(transaction: Transaction): Promise<RiskCheckResult> {
    try {
      return await this.circuitBreaker.execute(async () => {
        return await this.decisionEngine.checkRisk(transaction);
      });
    } catch (error) {
      // 熔断降级
      return { passed: false, reason: 'risk_check_unavailable' };
    }
  }
  
  private getHighestSeverity(anomalies: Anomaly[]): string {
    const severityOrder = ['low', 'medium', 'high', 'critical'];
    return anomalies
      .map(a => a.severity)
      .sort((a, b) => severityOrder.indexOf(b) - severityOrder.indexOf(a))[0];
  }
}
```

```typescript
// circuit-breaker.ts

export class CircuitBreaker {
  private state: CircuitBreakerState = CircuitBreakerState.CLOSED;
  private failures: number = 0;
  private lastFailureTime: number = 0;
  private successCount: number = 0;
  
  constructor(private config: CircuitBreakerConfig) {}
  
  /**
   * 执行受保护的函数
   */
  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === CircuitBreakerState.OPEN) {
      if (this.shouldAttemptRecovery()) {
        this.state = CircuitBreakerState.HALF_OPEN;
      } else {
        throw new CircuitBreakerOpenError('Circuit breaker is OPEN');
      }
    }
    
    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }
  
  /**
   * 判断是否应该尝试恢复
   */
  private shouldAttemptRecovery(): boolean {
    const now = Date.now();
    const timeSinceLastFailure = now - this.lastFailureTime;
    return timeSinceLastFailure >= this.config.behavior.recovery.recoveryTimeout * 1000;
  }
  
  /**
   * 成功回调
   */
  private onSuccess(): void {
    this.failures = 0;
    
    if (this.state === CircuitBreakerState.HALF_OPEN) {
      this.successCount++;
      if (this.successCount >= this.config.behavior.recovery.successThreshold) {
        this.state = CircuitBreakerState.CLOSED;
        this.successCount = 0;
      }
    }
  }
  
  /**
   * 失败回调
   */
  private onFailure(): void {
    this.failures++;
    this.lastFailureTime = Date.now();
    
    if (this.state === CircuitBreakerState.HALF_OPEN) {
      this.state = CircuitBreakerState.OPEN;
      this.successCount = 0;
    } else if (this.failures >= this.config.triggers.failureCount) {
      this.state = CircuitBreakerState.OPEN;
      this.triggerAlert();
    }
  }
  
  /**
   * 检查熔断器是否开启
   */
  isOpen(): boolean {
    return this.state === CircuitBreakerState.OPEN;
  }
  
  /**
   * 获取当前状态
   */
  getState(): CircuitBreakerState {
    return this.state;
  }
  
  /**
   * 手动触发告警
   */
  private triggerAlert(): void {
    if (this.config.alert.enabled) {
      // 发送告警通知
      console.error(`[CircuitBreaker] ${this.config.name} opened after ${this.failures} failures`);
    }
  }
  
  /**
   * 手动重置熔断器
   */
  reset(): void {
    this.state = CircuitBreakerState.CLOSED;
    this.failures = 0;
    this.successCount = 0;
  }
}
```

```typescript
// ai-disclosure.ts

export class AIDisclosure {
  private agentId: string;
  private operatorInfo: OperatorInfo;
  
  constructor(
    private config: DisclosureConfig,
    private agentName: string,
    private agentVersion: string
  ) {
    this.agentId = this.generateAgentId();
    this.operatorInfo = this.loadOperatorInfo();
  }
  
  /**
   * 生成 Agent ID
   */
  private generateAgentId(): string {
    const date = new Date().toISOString().slice(0, 10).replace(/-/g, '');
    const random = Math.random().toString(36).substring(2, 8);
    return `agent_${date}_${random}`;
  }
  
  /**
   * 加载运营方信息
   */
  private loadOperatorInfo(): OperatorInfo {
    return {
      name: '北京小龙虾科技有限公司',
      license: 'ICP证1234567号',
      contact: 'support@xiaolongxia.com'
    };
  }
  
  /**
   * 生成披露信息
   */
  generate(sessionId?: string): AIDisclosureInfo {
    return {
      version: '1.0',
      agentId: this.agentId,
      agentName: this.agentName,
      agentVersion: this.agentVersion,
      operator: {
        name: this.operatorInfo.name,
        license: this.operatorInfo.license,
        contact: this.operatorInfo.contact
      },
      generation: {
        model: 'gpt-4',
        timestamp: new Date().toISOString(),
        sessionId
      },
      disclosureType: this.config.compliance.showModel ? 'full' : 'compact',
      watermark: {
        enabled: true,
        algorithm: 'sha256',
        data: this.generateWatermark()
      }
    };
  }
  
  /**
   * 生成水印
   */
  private generateWatermark(): string {
    const data = `${this.agentId}:${Date.now()}`;
    return crypto.createHash('sha256').update(data).digest('hex');
  }
  
  /**
   * 渲染披露标识（用于消息展示）
   */
  renderDisclosure(disclosure: AIDisclosureInfo): string {
    const parts: string[] = [];
    
    // 图标和 Agent 标识
    parts.push(`[🤖 AI Agent: ${disclosure.agentName} v${disclosure.agentVersion}]`);
    
    // 标识 ID
    parts.push(`[标识ID: ${disclosure.agentId}]`);
    
    // 运营方
    if (this.config.compliance.showOperator) {
      parts.push(`[运营方: ${disclosure.operator.name}]`);
    }
    
    // 分隔线
    parts.push('─'.repeat(40));
    
    return parts.join('\n');
  }
  
  /**
   * 验证水印
   */
  verifyWatermark(content: string, watermark: string): boolean {
    const expected = this.generateWatermark();
    return expected === watermark;
  }
}
```
